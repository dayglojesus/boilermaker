    def process_options
      options.delete_if { |x,y| y.nil? }
      if options.empty?
        puts @optparse 
        exit 0
      end
      options.each do |x,y|
        begin
          if y.to_s.match('^-')
            raise BoilerMakerErr.new("Bad args: \"#{y}\" is not a valid arg to option, \"--#{x}\". Use the -h flag for syntax help.")
          end
        rescue => error
          puts error.message + "\n"
          exit 1
        end
      end
    end
