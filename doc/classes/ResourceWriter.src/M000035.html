  def write
    base = File.basename(@filein)
    prefix = "./resources/"
    fileout = "#{prefix + base.gsub(/thtml/, "html")}"
    # puts "Fileout: #{fileout}"
    puts "Generating: #{File.expand_path(fileout)}"
    target = File.open(fileout, 'w')
    target.puts @result
    target.close
  end
