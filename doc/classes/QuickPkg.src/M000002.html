  def initialize(source)
    @stamp = rand(100000)
    # Get rid of trailing "/" characters in the source path
    if source[:quickpkg] =~ /\/$/
      @source = source[:quickpkg].chop
    else
      @source = source[:quickpkg]
    end
    @pkginfo = {}
    @args = {}
    getvars(@source)
    procvars
  end
